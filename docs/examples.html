Change the first letter to lowercase

library(stringr)

fk_string <- c("Apple Pie", "Chocolate Cake")
str_sub(fk_string, 1, 1) <- str_to_lower(str_sub(fk_string, 1, 1)) 
fk_string
## [1] "apple Pie"      "chocolate Cake"

Removing a prefix

library(tidyverse)

fk_data <- 
  tibble(a_var = c("10   poor 1", 
                   "20  ok 2", 
                   "30  very good 3",
                   "999 NA"),
         x = rnorm(4))

rm_prefix_df <- 
  fk_data %>% 
  mutate(the_prefix = as.numeric(str_extract(a_var, "^\\d+"))) %>% 
  mutate(a_var = str_replace(a_var, "^(\\d+\\s+)", ""))

rm_prefix_df
## # A tibble: 4 x 3
##   a_var             x the_prefix
##   <chr>         <dbl>      <dbl>
## 1 poor 1       0.0640         10
## 2 ok 2        -0.107          20
## 3 very good 3 -1.38           30
## 4 NA           0.313         999

Removing a suffix

library(tidyverse)

fk_data <- 
  tibble(region = c("A region", 
                    "AB region", 
                    "ABC region"),
         x = rnorm(3))

## Method 1
rm_suffix_df <- 
  fk_data %>% 
  mutate(region = str_replace(region, "\\s(region)$", ""))

(rm_suffix_df)
## # A tibble: 3 x 2
##   region      x
##   <chr>   <dbl>
## 1 A      -0.424
## 2 AB      0.361
## 3 ABC     1.22
## Method 2
rm_suffix_df <- 
  fk_data %>%
  mutate(region = str_sub(region, 1, nchar(region) - 7))
(rm_suffix_df)
## # A tibble: 3 x 2
##   region      x
##   <chr>   <dbl>
## 1 A      -0.424
## 2 AB      0.361
## 3 ABC     1.22

Removing spaces

library(stringr)

x <- "   this  is  a    big    space problem.  "
(x <- str_trim(x, side = "both"))
## [1] "this  is  a    big    space problem."
(x <- str_replace_all(x, "\\s+", " "))
## [1] "this is a big space problem."

Using str_split_fixed

library(tidyverse)

fk_df <- 
  tibble(string = 
           c("NZTA: New Zealand Transport Agency",
             "PSU: Primary Sampling Unit",
             "NPD: Non Private Dwelling"))

split_matrix <- str_split_fixed(fk_df$string, ":", n = 2)

split_fk_df <- as.tibble(split_matrix)
names(split_fk_df) <- c("name", "description")
split_fk_df <- 
  split_fk_df %>% 
  mutate(description = str_trim(description, side = "left"))

split_fk_df
## # A tibble: 3 x 2
##   name  description                 
##   <chr> <chr>                       
## 1 NZTA  New Zealand Transport Agency
## 2 PSU   Primary Sampling Unit       
## 3 NPD   Non Private Dwelling

Difference between .+ and .*

library(tidyverse)

fk_data <-
  data.frame(x = c('123_123', 
                   '456_123', 
                   '567_23', 
                   '123_',

fk_data_a <-
